open /portal/public/classic/
echo -CategoryManagement-
click link=Sign in
type username root
type password gtn
clickAndWait //div[@id='UIPortalLoginFormAction']/div/div/div
echo --Select "Application Registry"
clickAt link=Application Registry 1,1
echo --Add Category
waitForElementPresent //div[@id='UIApplicationOrganizer']//div[@class='UIControlbar']/div[1] 1,1
clickAt //div[@id='UIApplicationOrganizer']//div[@class='UIControlbar']/div[1] 1,1
waitForElementPresent name
type name test_name_category_10
type displayName test_displayname_category_10
type description test_description_category_10
echo --Select permissions
click //div[@class='WorkingArea']//div[@class='TabsContainer']/div[2]//div[@class='MiddleTab']
waitForElementPresent link=Add Permission
clickAt link=Add Permission
waitForElementPresent link=Platform
clickAt link=Platform
waitForElementPresent link=manager
clickAt link=manager
clickAt link=Save
waitForElementPresent //div[@id='UIApplicationOrganizer']//a[@class='ControlIcon EditIcon']
waitForTextPresent test_displayname_category_10
verifyTextPresent test_displayname_category_10
echo --Edit Category
clickAt //div[@id='UIApplicationOrganizer']//a[@class='ControlIcon EditIcon'] 1,1
waitForElementPresent displayName
type displayName test_displayname_edit_10
clickAt link=Save
waitForTextPresent test_displayname_edit_10
verifyTextPresent test_displayname_edit_10
echo --Delete Category
clickAt //div[@class='SelectedTab']//a[@class='ControlIcon DeleteIcon']
waitForConfirmation Are you sure to delete this category and all applications on it?
waitForTextNotPresent test_displayname_edit_10
verifyTextNotPresent test_displayname_edit_10
waitForElementPresent link=Sign out
click link=Sign out
